5、完美数

#include
#define ll long long
using namespace std;
const int mod=1e9+7;
const int N=1e6+1;
int a,b;
ll fact[N]; //fact[i]表示i的阶乘
ll infact[N]; //infact[i]表示i的阶乘的逆元
bool check(ll x)
{
while(x>0)
{
if(x%10!=a&&x%10!=b)
{
return 0;
}
x/=10;
}
return 1;
}
int qmi(int a,int b) //求a的b次方对mod求模的结果
{
ll res=1;
while(b)
{
if(b&1)res=resa%mod;
a=a(ll)a%mod;
b>>=1;
}
return res;
}
void rr() //求阶乘及阶乘的逆元
{
fact[0]=1;
for(int i=1;i<N;i++)
{
fact[i]=fact[i-1]i%mod;
}
infact[N-1]=qmi(fact[N-1],mod-2);
for(int i=N-2;i>=0;i–)
{
infact[i]=infact[i+1](i+1)%mod;
}

}

int main()
{
std::ios::sync_with_stdio(false);
cin.tie(0);
cout.tie(0);
ll m;
ll res=0;
cin>>a>>b>>m;
rr();
if(a==b)
{
cout<<0;
return 0;
}
if(a>b)swap(a,b); //a作为小一点的数
for(ll i=0;i<=m;i++)
{
ll s=i*a+(m-i)*b;
if(check(s))
{ //将i个a和(m-i)个b组合成一个好数即可
ll i2=min(i,m-i); //c(m)(i)==c(m)(m-i)
res=(res+fact[m]infact[i2]%modinfact[m-i2]%mod)%mod;//即排列组合，共有C(m)(i2)种组合
}
}
cout<<res%mod;
return 0;

}
